```js
var removeElement = function(nums, val) {
    let left = 0, right = nums.length - 1;
    while(left <= right) {
        if (nums[right] === val) {
            right--;
        } else if (nums[left] !== val) {
            left++;
        } else {
            nums[left] = nums[right];
            left++;
            right--;
        }
```

```js
/**
 * @param {number} n
 * @return {number[][]}
 */
var generateMatrix = function(n) {
  const matrix = Array.from(Array(n), () => Array(n));
  let dir = 'top';
  let j = 0;
  let k = 0;
  for (let i = 1; i<= n*n; i++) {
    switch (dir) {
      case 'top': {
        if (k + 2 >= n || matrix[j][k+2] !== undefined) {
          dir = 'right'
        } 
        matrix[j][k++] = i;
        break;
      }
      case 'right': { 
        if (j + 2 >=  n || matrix[j+2][k] !== undefined) {
          dir = 'bottom'
        } 
        matrix[j++][k] = i;
        break;
      }
      case 'bottom': {
        if (k - 2 <  0 || matrix[j][k-2] !== undefined) {
          dir = 'left'
        } 
        matrix[j][k--] = i;
        break;
      }
      case 'left': {
        if (j - 2 <  0 || matrix[j-2][k] !== undefined) {
          dir = 'top'
        } 
        matrix[j--][k] = i;
        break;
      }
    }
  }
  return matrix;
};
```

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} val
 * @return {ListNode}
 */
var removeElements = function(head, val) {
  let virtualHead = { next: head };
  let preNode = virtualHead;
  let nextNode = head;
  while(nextNode) {
    if (nextNode.val === val) {
      let tempNode = nextNode;
      nextNode = tempNode.next;
      preNode.next = nextNode;
      tempNode.next = null;
    } else {
      preNode = preNode.next;
      nextNode = nextNode.next;
    }
  }
  return virtualHead.next;
};
```

```js
var isAnagram = function(s, t) {
  let maps = {};
  for (let i = s.length -1;i >=0; i--) {
    const count = maps[s[i]];
    if (count) {
      maps[s[i]]++
    } else {
      maps[s[i]] = 1;

    const count = maps[t[i]];
```

```js
var fourSumCount = function(nums1, nums2, nums3, nums4) {
  let n = nums1.length
  let map = new Map();
  for (let i = n - 1;i >=0;i--) {
    const n1 = nums1[i];
    for (let j = n - 1;j >=0;j--) {
      // 计算 nums1 + nums2 的所有情况
      let sum = n1 + nums2[j];
      map.set(sum, (map.get(sum) | 0) + 1);
    }
  }
  let count = 0;
  for (let i = n - 1;i >=0;i--) {
    const n3 = -nums3[i];
    for (let j = n - 1;j >=0;j--) {
      // - nums3 - nums4 所有情况，每一种情况和 map 里都有 map.get(n3 - nums4[j]) 种组合，所以 count 是这样子计算的
      if (map.has(n3 - nums4[j])) {
        count+= map.get(n3 - nums4[j]) || 0;
      }
    }
  }
  return count;
};
```

```js
var evalRPN = function(tokens) {
  const stack = []
  for (let i = 0; i < tokens.length; i++) {
    const char = tokens[i];
    switch (char) {
      case '+': {
        const right = stack.pop();
        const left = stack.pop();
        stack.push(left + right);
        break;
      }
      case '-': {
        const right = stack.pop();
        const left = stack.pop();
        stack.push(left - right);
        break;

        const left = stack.pop();
        stack.push(left * right);

        const left = stack.pop();
        const result = left / right

        break;
      }
      default: {
        stack.push(+char);
      }
    }
  }
```

```js
var levelOrder = function(root) {
  const queue = [root];
  const res = [];
  while (queue.length) {
    const size = queue.length;
    const temp = [];
    for (let i = 0; i < size; i++) {
      const top = queue.shift();
      if (top) {
        temp.push(top.val);
        queue.push(top.left);
        queue.push(top.right);
      }
    }
    temp.length && res.push(temp);
  }
  return res;
};
```

```js
function compare(left, right) {
  if (left === null && right !== null || left !== null && right === null) return false;
  if (left === null && right === null) return true;
  if (left.val !== right.val) return false;
  const outResult = compare(left.left, right.right);
  const innerResult = compare(left.right, right.left);
  return outResult && innerResult;
}
var isSymmetric = function(root) {
  if (!root) return true;
  return compare(root.left, root.right);
```

```js
  count += countNodes(root.left);
  count += countNodes(root.right);
  return count;
};
```
30. 二叉树：我平衡么？

因为需要判断二叉树高度，所以需要判断数字
```js
var getDepth = function(root) {
  if (!root) return 0;
  const leftDepth = getDepth(root.left);
  const rightDepth = getDepth(root.right);
  if (leftDepth === -1 || rightDepth === -1) return -1;

  return Math.abs(rightDepth - leftDepth) <= 1 ? 1 + Math.max(rightDepth, leftDepth) : -1; 
};
var isBalanced = function(root) {
  return getDepth(root) !== -1;
}
```
