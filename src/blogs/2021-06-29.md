[27. 移除元素](https://leetcode-cn.com/problems/remove-element/)
双指针交换元素
```js
var removeElement = function(nums, val) {
    let left = 0, right = nums.length - 1;
    while(left <= right) {
        if (nums[right] === val) {
            right--;
        } else if (nums[left] !== val) {
            left++;
        } else {
            nums[left] = nums[right];
            left++;
            right--;
        }
    }
    // 返回left 是因为最后一个 left 指向的元素必定是 nums[left] === val 或者是 left === nums.length;那么满足条件的正好有 left 个
    return left;
};
```

5. 螺旋矩阵
[59. 螺旋矩阵 II](https://leetcode-cn.com/problems/spiral-matrix-ii/)
按照 
top top  top right
top top right right
left left bottom right
left bottom bottom bottom
顺时针画
```js
/**
 * @param {number} n
 * @return {number[][]}
 */
var generateMatrix = function(n) {
  const matrix = Array.from(Array(n), () => Array(n));
  let dir = 'top';
  let j = 0;
  let k = 0;
  for (let i = 1; i<= n*n; i++) {
    switch (dir) {
      case 'top': {
        if (k + 2 >= n || matrix[j][k+2] !== undefined) {
          dir = 'right'
        } 
        matrix[j][k++] = i;
        break;
      }
      case 'right': { 
        if (j + 2 >=  n || matrix[j+2][k] !== undefined) {
          dir = 'bottom'
        } 
        matrix[j++][k] = i;
        break;
      }
      case 'bottom': {
        if (k - 2 <  0 || matrix[j][k-2] !== undefined) {
          dir = 'left'
        } 
        matrix[j][k--] = i;
        break;
      }
      case 'left': {
        if (j - 2 <  0 || matrix[j-2][k] !== undefined) {
          dir = 'top'
        } 
        matrix[j--][k] = i;
        break;
      }
    }
  }
  return matrix;
};
```

6. 移除链表元素
[203. 移除链表元素](https://leetcode-cn.com/problems/remove-linked-list-elements/)
因为头结点也是需要判断是否被删除的节点，所以我们需要设置一个虚拟头结点，用来记录结果链表的头结点
```js
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} val
 * @return {ListNode}
 */
var removeElements = function(head, val) {
  let virtualHead = { next: head };
  let preNode = virtualHead;
  let nextNode = head;
  while(nextNode) {
    if (nextNode.val === val) {
      let tempNode = nextNode;
      nextNode = tempNode.next;
      preNode.next = nextNode;
      tempNode.next = null;
    } else {
      preNode = preNode.next;
      nextNode = nextNode.next;
    }
  }
  return virtualHead.next;
};
```

[第707题：设计链表](https://leetcode-cn.com/problems/design-linked-list)

```js
var Node = function (val, next) {
  this.val = val || null;
  this.next = next || null;
}

var MyLinkedList = function() {
  this.head = new Node();
};

MyLinkedList.prototype.get = function(index) {
  let order = 0;
  let node = this.head;
  while (node && order <= index) {
    node = node.next;
    order++;
  }
  return node ? node.val : -1;
};

MyLinkedList.prototype.addAtHead = function(val) {
  let next = this.head.next;
  const node = new Node(val, next);
  this.head.next = node;
};

MyLinkedList.prototype.addAtTail = function(val) {
  let node = this.head;
  while (node.next) {
    node = node.next;
  }
  node.next = new Node(val);
};

MyLinkedList.prototype.addAtIndex = function(index, val) {
  let order = 0;
  let node = this.head;
  while (node.next && order < index) {
    node = node.next;
    order++;
  }
  let next = node.next;
  const newNode = new Node(val, next);
  node.next = newNode;
};

MyLinkedList.prototype.deleteAtIndex = function(index) {
  let order = 0;
  let node = this.head;
  while (node.next && order < index) {
    node = node.next;
    order++;
  }
  if (!node.next) return;
  let tempNode = node.next;
  node.next = tempNode.next;
  tempNode.next = null;
};
```

```js
var reverseList = function(head) {
  let current = head;
  let preNode = null;
  while (current) {
    let tempNode = current.next;
    current.next = preNode;
    preNode = current;
    current = tempNode;
  }
  return preNode;
```

10. 哈希表：可以拿数组当哈希表来用，但哈希值不要太大！
[242. 有效的字母异位词](https://leetcode-cn.com/problems/valid-anagram/)
```js
/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isAnagram = function(s, t) {
  let maps = {};
  for (let i = s.length -1;i >=0; i--) {
    const count = maps[s[i]];
    if (count) {
      maps[s[i]]++
    } else {
      maps[s[i]] = 1;
    }
  }
  for (let i = t.length - 1; i >= 0; i--) {
    const count = maps[t[i]];
    if (count) {
      maps[t[i]]--
    } else {
      return false;
    }
  }
  return Object.values(maps).every(v => v === 0);
};
```
11. 哈希表：今天你快乐了么？
[202. 快乐数](https://leetcode-cn.com/problems/happy-number/)
易知，非快乐数会出现无限循环的情况，那么我们就可以记录下每一次的结果，一旦出现循环就确定它是 false
```js
var square = function (n) {
  const s = (n + '').split('');
  return s.reduce((sum, num) => {
    num = +num;
    sum += num*num;
    return sum
  }, 0)
}

var isHappy = function(n) {
  const set = new Set([n]);
  while (n !== 1) {
    n = square(n);
    if (set.has(n)) {
      return false;
    } else {
      set.add(n);
    }
  }
  return true;
};
```
